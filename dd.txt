    #!/bin/bash

    # Przejdź do katalogu z aplikacją
    cd /inzynierka # Zmień na ścieżkę do katalogu Twojej aplikacji

    # Pobierz najnowszy kod z repozytorium Git
    git pull origin master # lub nazwa Twojej gałęzi

    # Zainstaluj zależności (jeśli używasz npm/yarn)
    npm install # lub yarn install

    # Zbuduj aplikację (jeśli to konieczne, np. dla Next.js)
    npm run build # lub yarn build

    # Restartuj usługę aplikacji (dostosuj komendę do sposobu uruchamiania aplikacji)
    pm2 restart inzynierka # Jeśli używasz PM2
    # systemctl restart your-app-service.service # Jeśli używasz systemd
    # service your-app-service restart # Jeśli używasz servicea asdasd
